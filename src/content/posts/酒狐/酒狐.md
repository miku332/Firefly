  ;(function resetOverlay() {
    var leftover = document.querySelector('body > #redirect-overlay');
    if (leftover) leftover.remove();
    var overlay = document.getElementById('redirect-overlay');
    if (overlay) overlay.classList.remove('active');
  })();
  document.addEventListener('astro:page-load', resetOverlay);
  document.addEventListener('swup:contentReplaced', resetOverlay);

  function startRedirect() {
    var overlay = document.getElementById('redirect-overlay');
    document.body.appendChild(overlay);
    overlay.classList.add('active');
    document.documentElement.style.overflow = 'hidden';
    document.body.style.overflow = 'hidden';
    setTimeout(function() {
      document.documentElement.style.overflow = '';
      document.body.style.overflow = '';
      var url = '/releases/?repo=LuoTianOrange%2FSTS2_WineFox&version=latest&open=1';
      if (window.swup) {
        window.swup.navigate(url);
      } else {
        window.location.href = url;
      }
    }, 1200);
  }
